话不多说上代码

HTML部分

<!-- 设置分页主体 -->
<div class="fen-ye">
    <!-- 设置内容 -->
    <span>上一页</span>
    <!-- 加 id 因为这个得单独控制 -->
    <span id="ben-ye">1</span>
    <a href="">1</a>
    <a href="">2</a>
    <a href="">3</a>
    <a href="">4</a>
    <a href="">...</a>
    <a href="">9</a>
    <!-- 加 id 因为这个得单独控制 -->
    <a href="" id="xia-ye">下一页</a>
</div>

CSS部分

 /* 分页主体 */
 .fen-ye {
     /* 使用弹性 */
     display: flex;
     /* 让内容居中 */
     justify-content: center;
     /* 当挤不下的时候换行 */
     flex-wrap: wrap;
 }
        
 /* 设置 .fen-ye 里面的所有 span标签和a标签 */
 .fen-ye span,
 a {
     /* 让文字居中 */
     text-align: center;
     /* 设置文字大小为20像素 */
     font-size: 20px;
     /* 加 1像素 实现 黑色 边框 */
     border: 1px solid black;
     /* 让元素之间上下为 0 左右为 10像素 */
     margin: 0 10px;
 }
        
 /* 设置 .fen-ye 里面的所有 a标签和 id 为 ben-ye 的元素 */
 .fen-ye a,
 #ben-ye {
     /* 设置宽高为 50像素 */
     width: 50px;
     height: 50px;
     /* 设置行高为 50像素 目的是让文字垂直居中 */
     line-height: 50px;
     /* 删除所有 a标签和 id 为 ben-ye元素 的下划线*/
     text-decoration: none;
     /* 设置文本颜色为黑色 */
     color: black;
 }
        
 /* 当鼠标移动到该属性上时 属性会做出变化 */
 .fen-ye a:hover {
     /* 文本颜色变成黑色 */
     color: black;
     /* 背景颜色变成 blueviolet 色 */
     background-color: blueviolet;
 }
        
 /* 设置 .fen-ye 里面的所有 span标签和 id 为 xia-ye 的元素 */
 .fen-ye span,
 #xia-ye {
     /* 设置宽为 100像素 */
     width: 100px;
     /* 设置行高为 50像素 目的是让文字垂直居中 */
     line-height: 50px;
 }
        
 /* 设置 id 为 ben-ye 的元素 目的是当前页面是第几页 */
 #ben-ye {
     /* 文本颜色为白色 */
     color: white;
     /* 背景颜色为 coral 色 */
     background-color: coral;
 }
